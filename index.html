2.HOLDER可以200SHARE的代價，設定/編輯自己喜歡的推薦碼(例如：Iamatiger)，但不可與其他HOLDER已存在的推薦碼重複。
3.推薦碼只允許(a~z,0~9,_,-)組合，字母不分大小寫。
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;
contract ReferralSystem {
mapping(address => string) public referralCodes; // 記錄每個 HOLDER 的推薦碼
mapping(string => address) public referralCodeOwners; // 記錄推薦碼對應的 HOLDER 地址
mapping(address => uint256) public balances; // 記錄 HOLDER 的 SHARE 代幣餘額
uint256 public constant SET_REFERRAL_CODE_COST = 200; // 設定/編輯推薦碼的代價為 200 SHARE
event ReferralCodeSet(address indexed holder, string referralCode);
event ReferralCodeUpdated(address indexed holder, string oldCode, string newCode);
/// @dev 設定或編輯推薦碼
function setMyReferralCode(string memory refCode) external {
// 確認推薦碼格式是否合法
require(isValidReferralCode(refCode), "Invalid referral code format");
// 轉換推薦碼為小寫，避免大小寫重複
string memory lowerRefCode = toLower(refCode);
// 確認推薦碼是否已存在
require(referralCodeOwners[lowerRefCode] == address(0), "Referral code already exists");
// 確認使用者有足夠的 SHARE 代幣
require(balances[msg.sender] >= SET_REFERRAL_CODE_COST, "Insufficient SHARE to set referral code");
// 如果使用者已經有推薦碼，刪除舊的推薦碼記錄
if (bytes(referralCodes[msg.sender]).length > 0) {
string memory oldCode = referralCodes[msg.sender];
delete referralCodeOwners[oldCode]; // 移除舊的推薦碼映射
// 發布事件，記錄推薦碼更新
emit ReferralCodeUpdated(msg.sender, oldCode, lowerRefCode);
} else {
// 發布事件，記錄新的推薦碼設定
emit ReferralCodeSet(msg.sender, lowerRefCode);
}
// 扣除設定推薦碼的費用
balances[msg.sender] -= SET_REFERRAL_CODE_COST;
// 設定新的推薦碼
referralCodes[msg.sender] = lowerRefCode;
referralCodeOwners[lowerRefCode] = msg.sender;
}
/// @dev 驗證推薦碼格式是否有效 (僅允許 a~z, 0~9 組合)
function isValidReferralCode(string memory refCode) public pure returns (bool) {
bytes memory codeBytes = bytes(refCode);
for (uint256 i = 0; i < codeBytes.length; i++) {
bytes1 char = codeBytes[i];
if (
(char < 0x30 || char > 0x39) && // 不是 '0'-'9'
(char < 0x41 || (char > 0x5A && char < 0x61) || char > 0x7A) &&
// 不是 'A'-'Z' 或 'a'-'z'
(char != 0x5F) && // 不是 '_'
(char != 0x2D) && // 不是 '-'
(char != 0x40) // 不是 '@'
) {
return false;
}
}
return true;
}
/// @dev 將推薦碼轉換為小寫
function toLower(string memory str) public pure returns (string memory) {
bytes memory strBytes = bytes(str);
for (uint256 i = 0; i < strBytes.length; i++) {
if (strBytes[i] >= 0x41 && strBytes[i] <= 0x5A) {
// 將大寫字元轉換為小寫
strBytes[i] = bytes1(uint8(strBytes[i]) + 32);
}
}
return string(strBytes);
}
}

三、HOLDER與其介紹人(Referral)關係的建立
1.HOLDER不可以介紹自己。
2.在接收到第一筆超過100個SHARE的當下，Sender如果不是被排除介紹人的地址，就會自動成為Holder的介紹人。
3.在擁有至少100個SHARE的HOLDER，但介紹人依然是空白的情況下，Holder可以手動填入介紹人的推薦碼，成為介紹人的下線。
// --- 自動建立 HOLDER 與介紹人的關係 ---
function transfer(address recipient, uint256 amount) public override returns (bool) {
// 呼叫父合約的轉帳邏輯
bool result = super.transfer(recipient, amount);
// 建立自動推薦邏輯
if (amount >= MIN_REFERRAL_AMOUNT && holders[recipient].referrer == address(0)) {
address sender = _msgSender();
// 確保 Sender 非合約地址且未被排除
if (!isContract(sender) && !isExcludedFromReferral[sender]) {
holders[recipient].referrer = sender;
emit ReferralSet(recipient, sender);
}
}
return result;
}
// --- 手動設定介紹人 —
function setReferrer(string memory refCode) public {
require(balanceOf(msg.sender) >= MIN_REFERRAL_AMOUNT, "Insufficient balance to set referrer");
require(holders[msg.sender].referrer == address(0), "Referrer already set");
require(bytes(refCode).length > 0, "Invalid referral code");
address referrer = referralCodeToAddress[refCode];

1.逾時不候，當任何一位HOLDER呼叫空投程序，發現空投時間

為了不讓HOLDER花費GAS卻沒有獲得任何空投，特設計檢視能否領取空投的程序：
//檢視HOLDER能否領取空投的程序
function checkAirdropEligibility() external view returns (bool eligible, string memory reason) {
if (!airdropEnabled) {
return (false, "Airdrop is not enabled");
}
if (block.timestamp > airDropStartTime + 72 hours) {
return (false, "Airdrop period has expired");
}
if (holders[msg.sender].hasClaimedAirdrop) {
return (false, "You have already claimed the airdrop");
}
// 使用字串串聯來建立新的訊息
string memory message = string(abi.encodePacked("You are eligible to claim the ", toString(airDropCount), "th airdrop"));
return (true, message);
}
// 將 uint256 轉換為 string 的函式
function toString(uint256 value) internal pure returns (string memory) {
// 最多可容納 78 個字元 (uint256 的最大值)
bytes memory buffer = new bytes(78); uint256 i = buffer.length;
if (value == 0) { return "0"; }
while (value > 0) { i--; buffer[i] = bytes1(uint8(value % 10) + 0x30);
value /= 10; }
return string(buffer[i..]);
}

// Claim airdrop
function claimAirdrop() external {
require(airdropEnabled, "Airdrop is not enabled");
require(block.timestamp <= airDropStartTime + airDropDuration, "Airdrop period has expired");
require(holders[msg.sender].lastClaimedAirDropTime < airDropStartTime, "You have already claimed the airdrop");
uint256 reward = (balanceOf(msg.sender) * 3) / 100; // 3% of holder's balance
uint256 referrerBonus = (balanceOf(msg.sender) * 1) / 100; // 1% to referrer
_mint(msg.sender, reward);
holders[msg.sender].lastClaimedAirDropTime = block.timestamp; // 更新上次領取時間
address referrer = referrers[msg.sender];
if (referrer != address(0)) {
_mint(referrer, referrerBonus);
}
emit AirdropClaimed(msg.sender, reward, referrer, referrerBonus);
}
